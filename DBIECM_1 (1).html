def update_cluster(cluster, x):
#     cluster is a dictinary
    cluster['points'].append(x)
    pts = np.array(cluster['points'])
    cluster['center'] = np.mean(pts, axis=0)
    # The new radius is the maximum distance from the updated center to any point in the cluster.
    cluster['radius'] = np.max(np.linalg.norm(pts - cluster['center'], axis=1))
    return cluster

def simulate_addition(cluster, x):
# just to see what would happen for DBI
    pts = np.array(cluster['points'] + [x])
    new_center = np.mean(pts, axis=0)
    new_radius = np.max(np.linalg.norm(pts - new_center, axis=1))
    return new_center, new_radius

def compute_dbi(clusters):

    k = len(clusters)
    if k < 2:
        return 0  # With one cluster DBI is defined as 0.
    
    dbi_values = []
    for i in range(k):
        max_ratio = -np.inf
        for j in range(k):
            if i == j:
                continue
            center_i = clusters[i]['center']
            center_j = clusters[j]['center']
            d_ij = np.linalg.norm(center_i - center_j)
            # Avoid division by zero: if centers are identical, treat the ratio as infinite.
            ratio = (clusters[i]['radius'] + clusters[j]['radius']) / d_ij if d_ij != 0 else np.inf
            if ratio > max_ratio:
                max_ratio = ratio
        dbi_values.append(max_ratio)
    return np.mean(dbi_values)

def run_DBIECM(data, r0, chunk_size=100):


    clusters = []  # list to store cluster dictionaries
    labels = []    # parallel list: for each data point, record the cluster index it was assigned to
    
    # Process the data in chunks:
    n_samples = data.shape[0]
    for start in range(0, n_samples, chunk_size):
        chunk = data[start:start+chunk_size]
        for x in chunk:
            x = np.array(x) 
            # If no clusters yet, create the very first cluster.
            if not clusters:
                clusters.append({'points': [x], 'center': x, 'radius': 0.0})
                labels.append(0)
                continue

            # Calculate the distance from x to each cluster center.
            distances = [euclidean_distance(x, cluster['center']) for cluster in clusters]

            # --- Condition 1 ---
            # If the new point falls within the radius of any cluster, assign it to one of these.
            # (Here we choose the candidate with the smallest distance.)
            candidate1 = [i for i, cluster in enumerate(clusters) if distances[i] <= cluster['radius']]
            if candidate1:
                best_idx = min(candidate1, key=lambda i: distances[i])
                update_cluster(clusters[best_idx], x)
                labels.append(best_idx)
                continue

            # --- Condition 2 ---
            # If x is farther than r0 from every cluster center, create a new cluster.
            if all(d > r0 for d in distances):
                clusters.append({'points': [x], 'center': x, 'radius': 0.0})
                labels.append(len(clusters) - 1)
                continue

            # --- Else Condition (Borderline case) ---
            # There exist clusters for which: cluster.radius < distance < r0.
            candidate2 = [i for i, d in enumerate(distances) if clusters[i]['radius'] < d < r0]
            best_dbi = np.inf
            best_idx = None
            for i_candidate in candidate2:
                # Simulate the effect of adding x to candidate cluster i_candidate.
                temp_clusters = []
                for j, cluster in enumerate(clusters):
                    if j == i_candidate:
                        new_center, new_radius = simulate_addition(cluster, x)
                        temp_clusters.append({
                            'points': cluster['points'] + [x],
                            'center': new_center,
                            'radius': new_radius
                        })
                    else:
                        # For other clusters, we simply use their current information.
                        temp_clusters.append(cluster)
                dbi_val = compute_dbi(temp_clusters)
                if dbi_val < best_dbi:
                    best_dbi = dbi_val
                    best_idx = i_candidate
            if best_idx is not None:
                update_cluster(clusters[best_idx], x)
                labels.append(best_idx)
            else:
                # Fallback: if no candidate was found (this should rarely occur), create a new cluster.
                clusters.append({'points': [x], 'center': x, 'radius': 0.0})
                labels.append(len(clusters) - 1)
    return clusters, labels
